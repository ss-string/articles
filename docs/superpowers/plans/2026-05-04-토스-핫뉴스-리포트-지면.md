# 토스 핫뉴스 리포트 지면 구현 계획

> **에이전트 작업자 필수 안내:** 이 계획을 작업 단위로 실행할 때는 `superpowers:subagent-driven-development` 사용을 권장하며, 대안으로 `superpowers:executing-plans`를 사용할 수 있다. 진행 상황은 체크박스(`- [ ]`)로 추적한다.

**목표:** Supabase `toss_wts_hot_news_reports` 데이터를 카드 지면으로 보여주고, 카드 선택 시 row 기반 리포트 팝업을 띄운다.

**구조:** 기존 컨센서스 기능처럼 데이터 조회, 정규화, 상태 훅, 화면 컴포넌트를 분리한다. `report_payload`를 우선 사용하고 상위 컬럼을 대체값으로 사용해 `HotNewsReport` 표시 모델을 만든다. 팝업은 독립 컴포넌트로 두고 딤드, 닫기 인터랙션, 링크 처리를 담당한다.

**기술 스택:** React 19, TypeScript, Vite, Vitest, Testing Library, Supabase JS.

```ts
export type RawHotNewsReportRow = Record<string, unknown>;

export type HotNewsArticle = {
  title: string;
  link: string | null;
};

export type HotNewsCompanyEvidence = {
  company: string;
  code: string | null;
  position: 'bull' | 'bear' | 'neutral' | string | null;
  evidence: string[];
  links: string[];
};

export type HotNewsReport = {
  id: string;
  issueDate: string | null;
  displayDate: string;
  title: string;
  perspective: string | null;
  tldr: string[];
  interpretation: string | null;
  keyArticles: HotNewsArticle[];
  companyEvidence: HotNewsCompanyEvidence[];
};

function isRecord(value: unknown): value is Record<string, unknown> {
  return Boolean(value) && typeof value === 'object' && !Array.isArray(value);
}

function parseText(value: unknown): string | null {
  if (typeof value !== 'string' && typeof value !== 'number') {
    return null;
  }

  const text = String(value).trim();
  return text.length > 0 ? text : null;
}

function readPayload(row: RawHotNewsReportRow): Record<string, unknown> {
  return isRecord(row.report_payload) ? row.report_payload : {};
}

function parseStringArray(value: unknown): string[] {
  if (!Array.isArray(value)) {
    return [];
  }

  return value
    .map(parseText)
    .filter((item): item is string => item !== null);
}

function parseArticles(value: unknown): HotNewsArticle[] {
  if (!Array.isArray(value)) {
    return [];
  }

  return value.flatMap((item) => {
    if (!isRecord(item)) {
      return [];
    }

    const title = parseText(item.title);
    if (!title) {
      return [];
    }

    return [{ title, link: parseText(item.link) }];
  });
}

function parseCompanyEvidence(value: unknown): HotNewsCompanyEvidence[] {
  if (!Array.isArray(value)) {
    return [];
  }

  return value.flatMap((item) => {
    if (!isRecord(item)) {
      return [];
    }

    const company = parseText(item.company);
    if (!company) {
      return [];
    }

    return [
      {
        company,
        code: parseText(item.code),
        position: parseText(item.position),
        evidence: parseStringArray(item.detailedEvidence),
        links: parseStringArray(item.detailedNewsLinks),
      },
    ];
  });
}

export function formatIssueDate(value: unknown): string {
  const text = parseText(value);
  if (!text) {
    return '-';
  }

  const match = text.match(/^(\\d{4})-(\\d{2})-(\\d{2})$/);
  if (!match) {
    return text;
  }

  return `${match[1]}.${match[2]}.${match[3]}`;
}

export function normalizeHotNewsReport(row: RawHotNewsReportRow): HotNewsReport | null {
  const payload = readPayload(row);
  const title = parseText(payload.title) ?? parseText(row.title);

  if (!title) {
    return null;
  }

  const id = parseText(row.id) ?? title;
  const issueDate = parseText(row.issue_date);

  return {
    id,
    issueDate,
    displayDate: formatIssueDate(issueDate),
    title,
    perspective: parseText(row.perspective),
    tldr: parseStringArray(payload.tldr).length > 0 ? parseStringArray(payload.tldr) : parseStringArray(row.tldr),
    interpretation: parseText(payload.interpretation) ?? parseText(row.interpretation),
    keyArticles:
      parseArticles(payload.keyArticles).length > 0
        ? parseArticles(payload.keyArticles)
        : parseArticles(row.key_articles),
    companyEvidence:
      parseCompanyEvidence(payload.companyNewsEvidence).length > 0
        ? parseCompanyEvidence(payload.companyNewsEvidence)
        : parseCompanyEvidence(row.company_news_evidence),
  };
}

export function buildHotNewsReports(rows: RawHotNewsReportRow[]): HotNewsReport[] {
  return rows
    .map(normalizeHotNewsReport)
    .filter((report): report is HotNewsReport => report !== null);
}
```

```ts
import { createClient } from '@supabase/supabase-js';
import type { RawHotNewsReportRow } from './model';

const tableName = 'toss_wts_hot_news_reports';

export function getHotNewsSupabaseConfig() {
  const url = import.meta.env.VITE_SUPABASE_URL;
  const publishableKey = import.meta.env.VITE_SUPABASE_PUBLISHABLE_KEY;

  if (!url || !publishableKey) {
    throw new Error('Supabase 환경변수가 설정되지 않았습니다.');
  }

  return { url, publishableKey };
}

export async function queryHotNewsReportRows(): Promise<RawHotNewsReportRow[]> {
  const { url, publishableKey } = getHotNewsSupabaseConfig();
  const supabase = createClient(url, publishableKey);
  const { data, error } = await supabase
    .from(tableName)
    .select('*')
    .order('issue_date', { ascending: false })
    .order('id', { ascending: false });

  if (error) {
    throw new Error(error.message);
  }

  return data ?? [];
}
```

생성 `src/hot-news/useHotNewsReports.ts`:

```ts
import { useEffect, useState } from 'react';
import { queryHotNewsReportRows } from './api';
import { buildHotNewsReports, type HotNewsReport, type RawHotNewsReportRow } from './model';

type HotNewsReportsState =
  | { status: 'loading'; reports: HotNewsReport[]; error: null }
  | { status: 'success'; reports: HotNewsReport[]; error: null }
  | { status: 'error'; reports: HotNewsReport[]; error: string };

type UseHotNewsReportsOptions = {
  queryRows?: () => Promise<RawHotNewsReportRow[]>;
};

export function useHotNewsReports(options: UseHotNewsReportsOptions = {}): HotNewsReportsState {
  const [state, setState] = useState<HotNewsReportsState>({
    status: 'loading',
    reports: [],
    error: null,
  });

  useEffect(() => {
    let isMounted = true;
    const loadRows = options.queryRows ?? queryHotNewsReportRows;

    async function load() {
      setState({ status: 'loading', reports: [], error: null });

      try {
        const rawRows = await loadRows();
        const reports = buildHotNewsReports(rawRows);

        if (isMounted) {
          setState({ status: 'success', reports, error: null });
        }
      } catch (error) {
        if (isMounted) {
          setState({
            status: 'error',
            reports: [],
            error: error instanceof Error ? error.message : '핫뉴스 리포트를 불러오지 못했습니다.',
          });
        }
      }
    }

    void load();

    return () => {
      isMounted = false;
    };
  }, [options.queryRows]);

  return state;
}
```

```tsx
import { useState } from 'react';
import type { RawHotNewsReportRow, HotNewsReport } from '../hot-news/model';
import { useHotNewsReports } from '../hot-news/useHotNewsReports';
import { HotNewsReportModal } from './HotNewsReportModal';

type HotNewsReportsPageProps = {
  queryRows?: () => Promise<RawHotNewsReportRow[]>;
};

export function HotNewsReportsPage({ queryRows }: HotNewsReportsPageProps) {
  const state = useHotNewsReports({ queryRows });
  const [selectedReport, setSelectedReport] = useState<HotNewsReport | null>(null);

  const statusContent =
    state.status === 'loading' ? (
      <div className="state-panel">핫뉴스 리포트를 불러오는 중입니다.</div>
    ) : state.status === 'error' ? (
      <div className="state-panel error">{state.error}</div>
    ) : state.reports.length === 0 ? (
      <div className="state-panel">표시할 핫뉴스 리포트가 없습니다.</div>
    ) : null;

  return (
    <section className="dashboard-section hot-news-section" id="hot-news" aria-labelledby="hot-news-title">
      <div className="section-heading">
        <span>Toss WTS Hot News Reports</span>
        <h2 id="hot-news-title">핫뉴스 리포트</h2>
      </div>
      <p className="hot-news-copy">
        토스증권 주요 뉴스 묶음을 카드로 훑고, 선택한 이슈의 리포트를 팝업에서 확인합니다.
      </p>

      {statusContent}

      {state.status === 'success' && state.reports.length > 0 ? (
        <div className="hot-news-grid">
          {state.reports.map((report) => (
            <button className="hot-news-card" key={report.id} type="button" onClick={() => setSelectedReport(report)}>
              <span className="hot-news-date">{report.displayDate}</span>
              {report.perspective ? <span className="hot-news-perspective">{report.perspective}</span> : null}
              <strong>{report.title}</strong>
              {report.tldr[0] ? <p>{report.tldr[0]}</p> : null}
            </button>
          ))}
        </div>
      ) : null}

      {selectedReport ? (
        <HotNewsReportModal report={selectedReport} onClose={() => setSelectedReport(null)} />
      ) : null}
    </section>
  );
}
```
