```typescript
export type RawRealEstateRow = Record<string, unknown>;

export type RawRealEstateTables = {
  interestTargets: RawRealEstateRow[];
  complexes: RawRealEstateRow[];
  pyeongOptions: RawRealEstateRow[];
  articles: RawRealEstateRow[];
  priceMetrics: RawRealEstateRow[];
};

export type RealEstatePriceMetric = {
  metricDate: string;
  actualAveragePrice: number | null;
  askingAveragePrice: number | null;
  actualMedianPrice: number | null;
  actualMinPrice: number | null;
  actualMaxPrice: number | null;
};

export type RealEstateArticle = {
  articleNumber: string;
  complexId: string;
  pyeongType: string;
  price: number;
  tradeType: string | null;
  buildingName: string | null;
  floorInfo: string | null;
  articleUrl: string | null;
  priceGapFromMedian: number | null;
  priceGapPercentFromMedian: number | null;
};

export type RealEstateInterestTarget = {
  id: string;
  complexId: string;
  pyeongType: string;
  complexName: string;
  complexUrl: string | null;
  displayPyeongName: string;
  exclusiveAreaM2: number | null;
  households: number | null;
  useApprovedDate: string | null;
  sortOrder: number;
  latestMetric: RealEstatePriceMetric | null;
  metricsSeries: RealEstatePriceMetric[];
  currentArticles: RealEstateArticle[];
  belowMedianArticles: RealEstateArticle[];
};

export type RealEstateDashboard = {
  targets: RealEstateInterestTarget[];
};

const targetColumns = {
  complexId: ['complex_id', 'complexId'],
  pyeongType: ['pyeong_type', 'pyeongType'],
  sortOrder: ['sort_order', 'sortOrder', 'display_order'],
};

const complexColumns = {
  complexId: ['complex_id', 'complexId', 'id'],
  name: ['complex_name', 'complexName', 'name'],
  url: ['complex_url', 'complexUrl', 'url'],
  households: ['households', 'household_count', 'total_households'],
  useApprovedDate: ['use_approved_at', 'use_approved_date', 'approval_date'],
};

const pyeongColumns = {
  complexId: ['complex_id', 'complexId'],
  pyeongType: ['pyeong_type', 'pyeongType'],
  name: ['display_pyeong_name', 'pyeong_name', 'name'],
  area: ['exclusive_area_m2', 'exclusive_area', 'supply_area_m2'],
};

const articleColumns = {
  articleNumber: ['article_number', 'articleNumber', 'id'],
  complexId: ['complex_id', 'complexId'],
  pyeongType: ['pyeong_type', 'pyeongType'],
  tradeType: ['trade_type', 'tradeType', 'deal_type'],
  price: ['price', 'deal_or_warrant_prc', 'asking_price'],
  buildingName: ['building_name', 'buildingName', 'dong_name'],
  floorInfo: ['floor_info', 'floorInfo', 'floor'],
  articleUrl: ['article_url', 'articleUrl', 'url'],
};

const metricColumns = {
  complexId: ['complex_id', 'complexId'],
  pyeongType: ['pyeong_type', 'pyeongType'],
  metricDate: ['metric_date', 'metricDate', 'collected_at', 'created_at'],
  actualAveragePrice: ['actual_average_price', 'actual_avg_price', 'trade_average_price', 'average_price'],
  askingAveragePrice: ['asking_average_price', 'asking_avg_price', 'article_average_price'],
  actualMedianPrice: ['actual_median_price', 'median_price', 'trade_median_price'],
  actualMinPrice: ['actual_min_price', 'min_price', 'trade_min_price'],
  actualMaxPrice: ['actual_max_price', 'max_price', 'trade_max_price'],
};

function readValue(row: RawRealEstateRow, candidates: readonly string[]) {
  for (const key of candidates) {
    if (Object.prototype.hasOwnProperty.call(row, key)) {
      return row[key];
    }
  }
  return null;
}

function parseText(value: unknown): string | null {
  if (typeof value !== 'string' && typeof value !== 'number') return null;
  const text = String(value).trim();
  return text.length > 0 ? text : null;
}

function parseNumber(value: unknown): number | null {
  if (typeof value === 'number' && Number.isFinite(value)) return value;
  if (typeof value === 'string') {
    const parsed = Number(value.replace(/,/g, '').trim());
    return Number.isFinite(parsed) ? parsed : null;
  }
  return null;
}

function keyOf(complexId: string, pyeongType: string) {
  return `${complexId}:${pyeongType}`;
}

function parseMetric(row: RawRealEstateRow): (RealEstatePriceMetric & { complexId: string; pyeongType: string }) | null {
  const complexId = parseText(readValue(row, metricColumns.complexId));
  const pyeongType = parseText(readValue(row, metricColumns.pyeongType));
  const metricDate = parseText(readValue(row, metricColumns.metricDate));
  if (!complexId || !pyeongType || !metricDate) return null;
  return {
    complexId,
    pyeongType,
    metricDate,
    actualAveragePrice: parseNumber(readValue(row, metricColumns.actualAveragePrice)),
    askingAveragePrice: parseNumber(readValue(row, metricColumns.askingAveragePrice)),
    actualMedianPrice: parseNumber(readValue(row, metricColumns.actualMedianPrice)),
    actualMinPrice: parseNumber(readValue(row, metricColumns.actualMinPrice)),
    actualMaxPrice: parseNumber(readValue(row, metricColumns.actualMaxPrice)),
  };
}

function parseArticle(row: RawRealEstateRow, median: number | null): RealEstateArticle | null {
  const articleNumber = parseText(readValue(row, articleColumns.articleNumber));
  const complexId = parseText(readValue(row, articleColumns.complexId));
  const pyeongType = parseText(readValue(row, articleColumns.pyeongType));
  const price = parseNumber(readValue(row, articleColumns.price));
  if (!articleNumber || !complexId || !pyeongType || price === null || price <= 0) return null;
  const gap = median === null ? null : price - median;
  return {
    articleNumber,
    complexId,
    pyeongType,
    price,
    tradeType: parseText(readValue(row, articleColumns.tradeType)),
    buildingName: parseText(readValue(row, articleColumns.buildingName)),
    floorInfo: parseText(readValue(row, articleColumns.floorInfo)),
    articleUrl: parseText(readValue(row, articleColumns.articleUrl)),
    priceGapFromMedian: gap,
    priceGapPercentFromMedian: gap === null || median <= 0 ? null : (gap / median) * 100,
  };
}

export function buildRealEstateDashboard(tables: RawRealEstateTables): RealEstateDashboard {
  const complexes = new Map(
    tables.complexes.flatMap((row) => {
      const id = parseText(readValue(row, complexColumns.complexId));
      return id ? [[id, row] as const] : [];
    }),
  );
  const pyeongs = new Map(
    tables.pyeongOptions.flatMap((row) => {
      const complexId = parseText(readValue(row, pyeongColumns.complexId));
      const pyeongType = parseText(readValue(row, pyeongColumns.pyeongType));
      return complexId && pyeongType ? [[keyOf(complexId, pyeongType), row] as const] : [];
    }),
  );
  const metricsByTarget = tables.priceMetrics
    .map(parseMetric)
    .filter((metric): metric is RealEstatePriceMetric & { complexId: string; pyeongType: string } => metric !== null)
    .reduce((map, metric) => {
      const key = keyOf(metric.complexId, metric.pyeongType);
      const list = map.get(key) ?? [];
      list.push(metric);
      map.set(key, list);
      return map;
    }, new Map<string, (RealEstatePriceMetric & { complexId: string; pyeongType: string })[]>());

  return {
    targets: tables.interestTargets
      .flatMap((targetRow) => {
        const complexId = parseText(readValue(targetRow, targetColumns.complexId));
        const pyeongType = parseText(readValue(targetRow, targetColumns.pyeongType));
        if (!complexId || !pyeongType) return [];
        const key = keyOf(complexId, pyeongType);
        const complex = complexes.get(complexId);
        const pyeong = pyeongs.get(key);
        const metricsSeries = (metricsByTarget.get(key) ?? [])
          .sort((a, b) => a.metricDate.localeCompare(b.metricDate))
          .map(({ complexId: _complexId, pyeongType: _pyeongType, ...metric }) => metric);
        const latestMetric = metricsSeries[metricsSeries.length - 1] ?? null;
        const median = latestMetric?.actualMedianPrice ?? null;
        const currentArticles = tables.articles
          .filter((articleRow) => {
            const tradeType = parseText(readValue(articleRow, articleColumns.tradeType));
            return (
              parseText(readValue(articleRow, articleColumns.complexId)) === complexId &&
              parseText(readValue(articleRow, articleColumns.pyeongType)) === pyeongType &&
              (!tradeType || tradeType === '매매' || tradeType.toLowerCase() === 'sale')
            );
          })
          .map((articleRow) => parseArticle(articleRow, median))
          .filter((article): article is RealEstateArticle => article !== null)
          .sort((a, b) => a.price - b.price);
        return [
          {
            id: key,
            complexId,
            pyeongType,
            complexName: parseText(complex ? readValue(complex, complexColumns.name) : null) ?? complexId,
            complexUrl: parseText(complex ? readValue(complex, complexColumns.url) : null),
            displayPyeongName: parseText(pyeong ? readValue(pyeong, pyeongColumns.name) : null) ?? pyeongType,
            exclusiveAreaM2: parseNumber(pyeong ? readValue(pyeong, pyeongColumns.area) : null),
            households: parseNumber(complex ? readValue(complex, complexColumns.households) : null),
            useApprovedDate: parseText(complex ? readValue(complex, complexColumns.useApprovedDate) : null),
            sortOrder: parseNumber(readValue(targetRow, targetColumns.sortOrder)) ?? 0,
            latestMetric,
            metricsSeries,
            currentArticles,
            belowMedianArticles:
              median === null ? [] : currentArticles.filter((article) => article.price <= median),
          },
        ];
      })
      .sort((a, b) => a.sortOrder - b.sortOrder || a.complexName.localeCompare(b.complexName, 'ko')),
  };
}

export function formatKoreanHousePrice(value: number | null): string {
  if (value === null || !Number.isFinite(value)) return '-';
  const roundedMan = Math.round(value / 10000);
  const eok = Math.floor(roundedMan / 10000);
  const man = roundedMan % 10000;
  if (eok > 0 && man > 0) return `${eok}억 ${man.toLocaleString('ko-KR')}만`;
  if (eok > 0) return `${eok}억`;
  return `${man.toLocaleString('ko-KR')}만`;
}
```

```typescript
import { createClient } from '@supabase/supabase-js';
import type { RawRealEstateTables } from './model';

const tableNames = {
  interestTargets: 'real_estate_interest_targets',
  complexes: 'real_estate_complexes',
  pyeongOptions: 'real_estate_complex_pyeong_options',
  articles: 'real_estate_articles',
  priceMetrics: 'real_estate_pyeong_price_metrics',
} as const;

export function getSupabaseConfig() {
  const url = import.meta.env.VITE_SUPABASE_URL;
  const publishableKey = import.meta.env.VITE_SUPABASE_PUBLISHABLE_KEY;

  if (!url || !publishableKey) {
    throw new Error('Supabase 환경변수가 설정되지 않았습니다.');
  }

  return { url, publishableKey };
}

export async function queryRealEstateTables(): Promise<RawRealEstateTables> {
  const { url, publishableKey } = getSupabaseConfig();
  const supabase = createClient(url, publishableKey);
  const [interestTargets, complexes, pyeongOptions, articles, priceMetrics] = await Promise.all([
    supabase.from(tableNames.interestTargets).select('*').order('sort_order', { ascending: true }),
    supabase.from(tableNames.complexes).select('*'),
    supabase.from(tableNames.pyeongOptions).select('*'),
    supabase.from(tableNames.articles).select('*'),
    supabase.from(tableNames.priceMetrics).select('*'),
  ]);

  for (const result of [interestTargets, complexes, pyeongOptions, articles, priceMetrics]) {
    if (result.error) {
      throw new Error(result.error.message);
    }
  }

  return {
    interestTargets: interestTargets.data ?? [],
    complexes: complexes.data ?? [],
    pyeongOptions: pyeongOptions.data ?? [],
    articles: articles.data ?? [],
    priceMetrics: priceMetrics.data ?? [],
  };
}
```

```typescript
import { useEffect, useState } from 'react';
import { queryRealEstateTables } from './api';
import { buildRealEstateDashboard, type RawRealEstateTables, type RealEstateDashboard } from './model';

type RealEstateState =
  | { status: 'loading'; dashboard: RealEstateDashboard; error: null }
  | { status: 'success'; dashboard: RealEstateDashboard; error: null }
  | { status: 'error'; dashboard: RealEstateDashboard; error: string };

type UseRealEstateTransactionsOptions = {
  queryTables?: () => Promise<RawRealEstateTables>;
};

const emptyDashboard: RealEstateDashboard = { targets: [] };

export function useRealEstateTransactions(options: UseRealEstateTransactionsOptions = {}): RealEstateState {
  const [state, setState] = useState<RealEstateState>({
    status: 'loading',
    dashboard: emptyDashboard,
    error: null,
  });

  useEffect(() => {
    let isMounted = true;

    async function load() {
```

```typescript
function buildPolyline(series: RealEstatePriceMetric[], key: 'actualAveragePrice' | 'askingAveragePrice') {
  const values = series.map((metric) => metric[key]).filter((value): value is number => value !== null);
  if (series.length === 0 || values.length === 0) return '';
  const min = Math.min(...values);
  const max = Math.max(...values);

  return series

      const x = 24 + (index / Math.max(series.length - 1, 1)) * 552;
```
