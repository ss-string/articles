```typescript
function normalizeChartPoint(value: unknown): RealEstateChartPoint | null {
  if (!value || typeof value !== 'object') {
    return null;
  }

  const row = value as { tradeDate?: unknown; trade_date?: unknown; dealPrice?: unknown; deal_price?: unknown };
  const tradeDate = parseText(row.tradeDate ?? row.trade_date);
  const dealPrice = parseNumber(row.dealPrice ?? row.deal_price);

  return tradeDate && dealPrice !== null ? { tradeDate, dealPrice } : null;
}

function normalizeActiveListingRange(value: unknown): RealEstateActiveListingRange {
  if (!value || typeof value !== 'object') {
    return { count: 0, minDealPrice: null, maxDealPrice: null };
  }

  const row = value as {
    count?: unknown;
    minDealPrice?: unknown;
    min_deal_price?: unknown;
    maxDealPrice?: unknown;
    max_deal_price?: unknown;
  };

  return {
    count: parseNumber(row.count) ?? 0,
    minDealPrice: parseNumber(row.minDealPrice ?? row.min_deal_price),
    maxDealPrice: parseNumber(row.maxDealPrice ?? row.max_deal_price),
  };
}

function normalizeChartData(value: unknown): RealEstateChartData {
  if (!value || typeof value !== 'object') {
    return {
      series: [],
      latestRealPrice: null,
      highestRealPrice: null,
      activeListingRange: { count: 0, minDealPrice: null, maxDealPrice: null },
    };
  }

  const row = value as {
    series?: unknown;
    markers?: { latestRealPrice?: unknown; highestRealPrice?: unknown };
    activeListingRange?: unknown;
  };

  return {
    series: Array.isArray(row.series)
      ? row.series.map(normalizeChartPoint).filter((point): point is RealEstateChartPoint => point !== null)
      : [],
    latestRealPrice: normalizeChartPoint(row.markers?.latestRealPrice),
    highestRealPrice: normalizeChartPoint(row.markers?.highestRealPrice),
    activeListingRange: normalizeActiveListingRange(row.activeListingRange),
  };
}
```

- [ ] **Step 4: 단지 단위 병합 구현**

`buildRealEstateDashboard`에서 관심 대상을 `mergeKey(target.complexId)`로 묶는 흐름은 유지하고, 병합된 평형들의 `chartData`를 다음 규칙으로 합친다.

```typescript
function mergeChartSeries(metrics: RealEstatePriceMetric[]): RealEstateChartPoint[] {
  const byKey = new Map<string, RealEstateChartPoint>();

  for (const metric of metrics) {
    for (const point of metric.chartData.series) {
      byKey.set(`${point.tradeDate}:${point.dealPrice}`, point);
    }
  }

  return [...byKey.values()].sort((left, right) => {
    const dateCompare = left.tradeDate.localeCompare(right.tradeDate);
    return dateCompare !== 0 ? dateCompare : left.dealPrice - right.dealPrice;
  });
}

function getLatestRealPrice(series: RealEstateChartPoint[]) {
  return [...series].sort((left, right) => {
    const dateCompare = right.tradeDate.localeCompare(left.tradeDate);
    return dateCompare !== 0 ? dateCompare : right.dealPrice - left.dealPrice;
  })[0] ?? null;
}

function getHighestRealPrice(series: RealEstateChartPoint[]) {
  return [...series].sort((left, right) => right.dealPrice - left.dealPrice)[0] ?? null;
}

function mergeActiveListingRange(
  metrics: RealEstatePriceMetric[],
  articles: RealEstateArticle[],
): RealEstateActiveListingRange {
  const values = [
    ...metrics.flatMap((metric) => [
      metric.chartData.activeListingRange.minDealPrice,
      metric.chartData.activeListingRange.maxDealPrice,
    ]),
    ...articles.map((article) => article.price),
  ].filter((value): value is number => value !== null && Number.isFinite(value));

  return {
    count: articles.length,
    minDealPrice: values.length > 0 ? Math.min(...values) : null,
    maxDealPrice: values.length > 0 ? Math.max(...values) : null,
  };
}
```

```typescript
type ChartRenderPoint = {
  x: number;
  y: number;
  tradeDate: string;
  dealPrice: number;
};

function getChartDomain(target: RealEstateInterestTarget) {
  const values = [
    ...target.chartSeries.map((point) => point.dealPrice),
    target.activeListingRange.minDealPrice,
    target.activeListingRange.maxDealPrice,
  ].filter((value): value is number => value !== null && Number.isFinite(value));

  if (values.length === 0) {
    return null;
  }

  const min = Math.min(...values);
  const max = Math.max(...values);
  const range = max === min ? Math.max(max * 0.06, 1) : max - min;

  return {
    min: min - range * 0.12,
    max: max + range * 0.12,
  };
}

function buildChartPoints(target: RealEstateInterestTarget): ChartRenderPoint[] {
  const domain = getChartDomain(target);
  if (!domain) {
    return [];
  }

  const width = 560;
  const height = 270;
  const plot = { left: 44, right: 486, top: 58, bottom: 196 };
  const latestDate = new Date(`${target.updatedAt?.slice(0, 10) ?? target.latestRealPrice?.tradeDate ?? new Date().toISOString().slice(0, 10)}T00:00:00`);
  const startDate = new Date(latestDate);
  startDate.setDate(startDate.getDate() - 90);
  const windowMs = latestDate.getTime() - startDate.getTime();

  return target.chartSeries.map((point) => {
    const date = new Date(`${point.tradeDate}T00:00:00`);
    const ratio = windowMs <= 0 ? 0 : Math.max(0, Math.min(1, (date.getTime() - startDate.getTime()) / windowMs));

    return {
      ...point,
      x: plot.left + ratio * (plot.right - plot.left),
      y: plot.bottom - ((point.dealPrice - domain.min) / (domain.max - domain.min)) * (plot.bottom - plot.top),
    };
  });
}
```
