```ts
function parseBoolean(value: unknown): boolean {
  return value === true;
}

function parseNumber(value: unknown): number {
  if (typeof value === 'number' && Number.isFinite(value)) {
    return value;
  }
  if (typeof value === 'string' && value.trim() !== '') {
    const parsed = Number(value);
    return Number.isFinite(parsed) ? parsed : 0;
  }
  return 0;
}

function parseRecord(value: unknown): Record<string, string> {
  if (!isRecord(value)) {
    return {};
  }

  return Object.fromEntries(
    Object.entries(value)
      .map(([key, entry]) => [key, parseText(entry)])
      .filter((entry): entry is [string, string] => entry[1] !== null),
  );
}

export function stripIssueDatePrefix(value: string): string {
  return value.replace(/^\d{4}-\d{2}-\d{2}\s+/, '').trim();
}

export function formatKoreanDateTime(value: unknown): string | null {
  const text = parseText(value);
  if (!text) {
    return null;
  }

  const date = new Date(text);
  if (Number.isNaN(date.getTime())) {
    return text;
  }

  const parts = new Intl.DateTimeFormat('en-CA', {
    timeZone: 'Asia/Seoul',
    year: 'numeric',
    month: '2-digit',
    day: '2-digit',
    hour: '2-digit',
    minute: '2-digit',
    hourCycle: 'h23',
  }).formatToParts(date);
  const values = Object.fromEntries(parts.map((part) => [part.type, part.value]));
  return `${values.year}-${values.month}-${values.day} ${values.hour}:${values.minute}`;
}

export function getHotNewsChangeStatusLabel(status: HotNewsChangeStatus | null): string {
  switch (status) {
    case 'initial':
      return '초기 문서';
    case 'refresh':
      return '새로고침 업데이트';
    case 'material_change':
      return '중요 변경 업데이트';
    case 'deduplicated':
      return '중복 정리됨';
    default:
      return status ?? '상태 없음';
  }
}
```

```ts
const latestReportsViewName = 'toss_wts_hot_news_latest_reports';
const reportsTableName = 'toss_wts_hot_news_reports';

function createHotNewsClient() {
  const { url, publishableKey } = getHotNewsSupabaseConfig();
  return createClient(url, publishableKey);
}

export async function queryHotNewsReportRows(issueDate?: string): Promise<RawHotNewsReportRow[]> {
  const supabase = createHotNewsClient();
  let query = supabase
    .from(latestReportsViewName)
    .select('*')
    .order('issue_date', { ascending: false })
    .order('perspective_key', { ascending: true });

  if (issueDate) {
    query = query.eq('issue_date', issueDate);
  }

  const { data, error } = await query;
  if (error) {
    throw new Error(error.message);
  }
  return data ?? [];
}

export async function queryHotNewsLatestIssueDate(): Promise<string | null> {
  const supabase = createHotNewsClient();
  const { data, error } = await supabase
    .from(latestReportsViewName)
    .select('issue_date')
    .order('issue_date', { ascending: false })
    .limit(1);

  if (error) {
    throw new Error(error.message);
  }

  const firstRow = data?.[0] as RawHotNewsReportRow | undefined;
  return typeof firstRow?.issue_date === 'string' ? firstRow.issue_date : null;
}

export async function queryHotNewsReportHistoryRows(
  issueDate: string,
  perspectiveKey: string,
): Promise<RawHotNewsReportRow[]> {
  const supabase = createHotNewsClient();
  const { data, error } = await supabase
    .from(reportsTableName)
    .select('*')
    .eq('issue_date', issueDate)
    .eq('perspective_key', perspectiveKey)
    .order('is_latest', { ascending: false })
    .order('run_slot', { ascending: true });

  if (error) {
    throw new Error(error.message);
  }

  return data ?? [];
}
```
